---
category: offmethod
heading: off(name, fn, [useCapture])
---

Removes the previously added event listeners from each element in the collection, and returns the collection. 

    var rect = Pablo.rect();
    
    function print () {
      alert("triggered");
    }

    rect.on('foo', print);
    rect.trigger('foo');
    rect.off('foo', print);
    rect.trigger('foo');
